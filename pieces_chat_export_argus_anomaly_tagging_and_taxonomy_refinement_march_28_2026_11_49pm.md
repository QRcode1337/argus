```rust
/// Compute ∇FTLE using central differencing with Gaussian pre-smoothing.
/// Respects the 153-point minimum constraint on the field array.
pub fn compute_ftle_gradient(
    ftle_field: &Array1<f64>,
    grid_spacing: f64,
    sigma: f64,       // Gaussian smoothing σ (recommended: 2.0–3.0)
    clamp_max: f64,   // Max gradient magnitude (prevents NaN propagation)
) -> Result<Array1<f64>, PhantomError> {
    let n = ftle_field.len();
    
    // Enforce 153-point minimum (your Lab slider constraint)
    if n < 153 {
        return Err(PhantomError::InsufficientPoints {
            required: 153,
            provided: n,
        });
    }

    // Step 1: Gaussian smoothing to suppress noise before differentiation
    let smoothed = gaussian_smooth_1d(ftle_field, sigma)?;

    // Step 2: Central differencing (interior points)
    let mut gradient = Array1::<f64>::zeros(n);
    for i in 1..(n - 1) {
        let raw = (smoothed[i + 1] - smoothed[i - 1]) / (2.0 * grid_spacing);
        // Clamp to prevent NaN/Inf propagation into SONA
        gradient[i] = raw.clamp(-clamp_max, clamp_max);
    }

    // Forward/backward differencing at boundaries
    gradient[0] = ((smoothed[1] - smoothed[0]) / grid_spacing).clamp(-clamp_max, clamp_max);
    gradient[n - 1] = ((smoothed[n - 1] - smoothed[n - 2]) / grid_spacing).clamp(-clamp_max, clamp_max);

    Ok(gradient)
}

/// Gaussian 1D smoothing kernel
fn gaussian_smooth_1d(field: &Array1<f64>, sigma: f64) -> Result<Array1<f64>, PhantomError> {
    let radius = (3.0 * sigma).ceil() as usize;
    let kernel_size = 2 * radius + 1;
    
    // Build kernel
    let mut kernel = Array1::<f64>::zeros(kernel_size);
    let mut sum = 0.0;
    for i in 0..kernel_size {
        let x = i as f64 - radius as f64;
        kernel[i] = (-0.5 * (x / sigma).powi(2)).exp();
        sum += kernel[i];
    }
    kernel /= sum; // Normalize

    // Convolve (with boundary reflection)
    let n = field.len();
    let mut smoothed = Array1::<f64>::zeros(n);
    for i in 0..n {
        let mut val = 0.0;
        for j in 0..kernel_size {
            let idx = i as isize + j as isize - radius as isize;
            let clamped = idx.clamp(0, (n - 1) as isize) as usize;
            val += field[clamped] * kernel[j];
        }
        smoothed[i] = val;
    }
    Ok(smoothed)
}
```

```rust
/// Emit FTLE gradient events over the WebSocket telemetry channel.
```

### Correlation Query (Updated from Our Prior Design)

```sql
-- Correlate FTLE gradient spikes with PNEUMA response latency
-- and SONA routing decisions within a ±30s/+60s window
SELECT 
    f.emitted_at            AS ftle_spike_time,
    f.field                 AS ftle_field,
    f.ftle_value,
    f.gradient_mag,
    f.gradient_dir,
    f.regime_label,
    p.responded_at          AS pneuma_response_time,
    p.latency_ms,
    p.context,
    p.model_used,
    s.model_selected        AS sona_routed_model,
    s.routing_confidence,
    s.reservoir_spectral_radius
FROM ftle_events f
LEFT JOIN pneuma_latency p 
    ON p.responded_at BETWEEN f.emitted_at - INTERVAL '30 seconds' 
                          AND f.emitted_at + INTERVAL '60 seconds'
LEFT JOIN sona_observations s
    ON s.observed_at BETWEEN f.emitted_at - INTERVAL '5 seconds'
                         AND f.emitted_at + INTERVAL '5 seconds'
WHERE f.gradient_mag > 0.15
ORDER BY f.emitted_at DESC
LIMIT 500;
```
